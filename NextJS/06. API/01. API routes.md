// api/isLogin.ts
// path example : localhost:3000/api/isLogin?id=3
// response : { id: "3" } 

import type { NextApiRequest, NextApiResponse } from "next";

type Data = {
  id: string | string[];
};

export default function handler(
  req: NextApiRequest,
  res: NextApiResponse<Data>
) {
  const { id } = req.query;
  res.status(200).json({ id: id });
}
