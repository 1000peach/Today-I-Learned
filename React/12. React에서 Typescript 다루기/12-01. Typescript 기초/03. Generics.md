```typescript
class Queue<T> {
  list: T[] = [];

  get length() {
    return this.list.length;
  }

  enqueue(item: T) {
    this.list.push(item);
  } // item:T -> list: T[] push

  dequeue() {
    return this.list.shift();
  }
}

const queue = new Queue<number>();
// queue.list: number[]

queue.enqueue(0);
queue.enqueue(1);
queue.enqueue(2);
queue.enqueue(3);
queue.enqueue(4);

while (queue.length > 0) {
  console.log(queue.dequeue());
}
```
